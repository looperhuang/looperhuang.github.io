滑動視窗(Sliding Window)

發表於 2025-01-20 更新於 2025-04-25 分類於 基礎

概述

主要就是操作一個視窗的左右邊界，去檢查範圍內的數據，適合用於 Array 或是 String
可以避免重複已經檢查過的數據，優化時間複雜度，但只適用於連續數據

簡易範例

再給定的數組[2, 5, 8 ,6 ,1]中，找出連續 3 個數字最大總和

var arr = [2, 5, 8, 6, 1];
var findThreeLargest = function (arr) {
  let curSum = 0;
  // 初始視窗總和(前3個數總和)
  for (let i = 0; i < 3; i++) {
    curSum += arr[i];
  }

  let maxSum = curSum; // 最大總和
  let maxSumIndex = 0; // 最大總和的起始位置
  for (let i = 3; i < arr.length; i++) {
    curSum = curSum - arr[i - 3] + arr[i];
    // 更新最大總和的起始位置
    if (curSum > maxSum) {
      maxSum = curSum;
      maxSumIndex = i - 3;
    }
  }
  return { maxSum, maxSumIndex };
};
