時間複雜度

發表於 2024-11-27 更新於 2025-04-25 分類於 基礎

概述

時間複雜度代表的是演算法隨著資料量變大時的增長趨勢
數學符號大O記號(big-O notation)

常見型別

  • 常數階 O(1)
int Constant(int n)
{
    int count = 0;
    // 迴圈次數與 n 無關
    int size = 10000;
    for (int i = 0; i < size; i++)
    {
        count++;
    }
    return count;
}
  • 線性階 O(n)
int Linear(int n)
{
    int count = 0;
    // 迴圈次數 n 次
    for (int i = 0; i < n; i++)
    {
        count++;
    }
    return count;
}
  • 平方階 O(n2n^2n2)
int Quadratic(int n)
{
    int count = 0;
    // 迴圈次數 n*n 次
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < n; j++)
        {
            count++;
        }
    }
    return count;
}
  • 指數階 O(2n2^n2n)
int Exponential(int n)
{
    int count = 0;
    int bas = 1;
    for (int i = 0; i < n; i++)
    {
        //迴圈次數為2^n次
        for (int j = 0; j < bas; j++)
        {
            count++;
        }
        bas *= 2;
    }
    return count;
}
  • 對數階 O(logn)
int Logarithmic(int n)
{
    int count = 0;
    while (n > 1) {
        n /= 2;
        count++;
    }
    return count;
}
  • 線性對數階 O(nlogn)
int LinearLogRecur(int n)
{
    if (n <= 1) return 1;
    int count = LinearLogRecur(n / 2);
    for (int i = 0; i < n; i++)
    {
        count++;
    }
    return count;
}
  • 階乘階 O(n!)
int FactorialRecur(int n)
{
    if (n == 0) return 1;
    int count = 0;
    // n * (n - 1) * (n - 2) * ... * 2 * 1
    for (int i = 0; i < n; i++)
    {
        count += FactorialRecur(n - 1);
    }
    return count;
}
